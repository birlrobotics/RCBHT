1%%************************* Documentation *********************************
  2% snapData3:
  3% 1) Loads the force-torque data from file. 
  4% 2) Not in online implementation: Plots the data in a single plot if plotOptions = 0, or subplots if
  5% plotOptions = 1. 
  6% All plots are ajusted for best viewing peformance:
  7% - TOP & BOTTOM max limits
  8% - Duration of plots
  9%
 10% Input Parameters:
 11% StrategyType:     - Changes code according to strategy type, ie. 'HSA'
 12% FolderName:       - Folder where data is saved
 13% plotOptions       - tells whether plots should be graphed in subplots or
 14%                     not.
 15%**************************************************************************
 16function [fPath,StratTypeFolder,forceData,stateData,handles,TOP_LIMIT,BOTTOM_LIMIT]...
 17                                        =snapData3(StrategyType,FolderName,plotOptions)
 18
 19%% INTIALIZATION
 20    global DB_PLOT;         % Declared in snapVerification. Enables plotting.
 21    
 22    % Switch Flag
 23    SWITCH = 1;             % Used to determine whether to turn on margin's around plots for extra space when adjusting. 
 24    
 25%% Assing appropriate directoy based on Ctrl Strategy to read data files
 26    StratTypeFolder = AssignDir(StrategyType);
 27    
 28    if(ispc)
 29        fPath = 'C:\\Documents and Settings\\suarezjl\\My Documents\\School\\Research\\AIST\\Results';
 30    else
 31        fPath = '/home/grxuser/Documents/School/Research/AIST/Results/';
 32        % QNX    % '\\home\\hrpuser\forceSensorPlugin_Pivot\Results'; 
 33    end
 34
 35%% Load the data
 36    % Angle data, force data, joint spring data, state data
 37    [angleData,forceData,jointsnapData,stateData] = loadData(fPath,StratTypeFolder,FolderName);
 38    
 39    %----------------------------------------------------------------------
 40    % The following code plots the data in the Torques.dat file. If running
 41    % an online system, do not print.
 42    %----------------------------------------------------------------------
 43    if(DB_PLOT)    
 44        %% Plots
 45            if(DB_PLOT)
 46                % Different plots may have different durations. The duration in seconds
 47                % is hard-coded in this function for specific trials. 
 48                % A percentage is returned to use in scaling the plot axis. 
 49                [TIME_LIMIT_PERC, SIGNAL_THRESHOLD] = CustomizePlotLength(StrategyType,FolderName,forceData);   
 50            else
 51                TIME_LIMIT_PERC = -1; SIGNAL_THRESHOLD = -1;
 52            end
 53
 54        %% For PA10 include spring joint angles
 55        if(~strcmp(StrategyType,'HSA') && ~strcmp(StrategyType,'ErrorCharac'))
 56
 57        %% Plot Rotation Spring Joint Position
 58            if(plotOptions)
 59                pSJ1=subplot(4,2,1); plot(jointsnapData(:,1),jointsnapData(:,2:3) );
 60            else
 61                figure(1); plot(jointsnapData(:,1),jointsnapData(:,2:3) );
 62                pSJ1 = gca;
 63            end
 64            title('Snap Joint Position'); xlabel('Time (secs)'); ylabel('Joint Angle'); 
 65
 66            % Adjust Axes
 67            MARGIN = SWITCH;    
 68            AVERAGE = 0;
 69            if(DB_PLOT)
 70                [TOP_LIMIT_S1, BOTTOM_LIMIT_S1] = adjustAxes('Rotation Spring',jointsnapData,TIME_LIMIT_PERC,SIGNAL_THRESHOLD,MARGIN,AVERAGE);         
 71            else
 72                TOP_LIMIT_S1 = -1; BOTTOM_LIMIT_S1 = -1;
 73            end
 74
 75        %%  Repeat
 76            if(plotOptions)
 77                pSJ2=subplot(4,2,2); plot(jointsnapData(:,1),jointsnapData(:,2:3) );
 78            else
 79                figure(2); plot(jointsnapData(:,1),jointsnapData(:,2:3) );
 80                pSJ2=gca;
 81            end
 82            title('Snap Joint Position'); xlabel('Time (secs)'); ylabel('Joint Angle'); 
 83
 84            % Adjust Axes
 85            MARGIN = SWITCH;
 86            AVERAGE = 0;
 87            [TOP_LIMIT_S2, BOTTOM_LIMIT_S2] = adjustAxes('Rotation Spring',jointsnapData,TIME_LIMIT_PERC,SIGNAL_THRESHOLD,MARGIN,AVERAGE);  
 88        end
 89
 90%% For HIRO no spring joint angles in the camera
 91%% Plot Fx
 92            if(plotOptions==1)
 93                pFx=subplot(3,2,1); plot(forceData(:,1),forceData(:,2));
 94            else
 95                pFx=plot(forceData(:,1),forceData(:,2));
 96            end
 97            title('Fx Plot'); xlabel('Time (secs)'); ylabel('Force (N)');
 98
 99            % Adjust Axes
100            MARGIN = SWITCH;
101            AVERAGE = 0;    % uses the average value of data to compute the axis limits. useful for data with big impulses
102            [TOP_LIMIT_Fx, BOTTOM_LIMIT_Fx] = adjustAxes('Fx',forceData,StrategyType,TIME_LIMIT_PERC,SIGNAL_THRESHOLD,MARGIN,AVERAGE);    
103
104%% Plot Fy
105            if(plotOptions==1)
106                pFy=subplot(3,2,3); plot(forceData(:,1),forceData(:,3));
107            else
108                pFy=plot(forceData(:,1),forceData(:,3));
109            end
110            title('Fy Plot'); xlabel('Time (secs)'); ylabel('Force (N)');
111
112            % Adjust Axes
113            MARGIN = SWITCH;
114            AVERAGE = 0;        
115            [TOP_LIMIT_Fy, BOTTOM_LIMIT_Fy] = adjustAxes('Fy',forceData,StrategyType,TIME_LIMIT_PERC,SIGNAL_THRESHOLD,MARGIN,AVERAGE);    
116
117%% Plot Fz
118            if(plotOptions==1)    
119                pFz=subplot(3,2,5); plot(forceData(:,1),forceData(:,4));
120            else
121                pFz=plot(forceData(:,1),forceData(:,4));
122            end
123            title('Fz Plot'); xlabel('Time (secs)'); ylabel('Force (N)');
124
125            % Adjust Axes
126            MARGIN = SWITCH;
127            AVERAGE = 0;    
128            [TOP_LIMIT_Fz, BOTTOM_LIMIT_Fz] = adjustAxes('Fz',forceData,StrategyType,TIME_LIMIT_PERC,SIGNAL_THRESHOLD,MARGIN,AVERAGE);
129
130%% Plot Mx
131            if(plotOptions==1)    
132                pMx=subplot(3,2,2); plot(forceData(:,1),forceData(:,5));
133            else
134                pMx=plot(forceData(:,1),forceData(:,5));
135            end
136            title('Mx Plot'); xlabel('Time (secs)'); ylabel('Moment (N-m)');
137
138            % Adjust Axes
139            MARGIN = SWITCH;     % If you want to insert a margin into the plots, set true.
140            AVERAGE = 0;    % If you want to average the signal value
141            [TOP_LIMIT_Mx, BOTTOM_LIMIT_Mx] = adjustAxes('Mx',forceData,StrategyType,TIME_LIMIT_PERC,SIGNAL_THRESHOLD,MARGIN,AVERAGE);
142
143
144%% Plot My
145            if(plotOptions==1)    
146                pMy=subplot(3,2,4); plot(forceData(:,1),forceData(:,6));
147            else
148                pMy=plot(forceData(:,1),forceData(:,6));
149            end
150            title('My Plot'); xlabel('Time (secs)'); ylabel('Moment (N-m)');
151
152            % Adjust Axes
153            MARGIN = SWITCH;
154            AVERAGE = 0;    
155            [TOP_LIMIT_My, BOTTOM_LIMIT_My] = adjustAxes('My',forceData,StrategyType,TIME_LIMIT_PERC,SIGNAL_THRESHOLD,MARGIN,AVERAGE);
156
157%% Plot Mz
158            if(plotOptions==1)    
159                pMz=subplot(3,2,6); plot(forceData(:,1),forceData(:,7));
160            else
161                pMz=plot(forceData(:,1),forceData(:,7));
162            end
163            title('Mz Plot'); xlabel('Time (secs)'); ylabel('Moment (N-m)');
164
165            % Adjust Axes
166            MARGIN = SWITCH;
167            AVERAGE = 0;
168            [TOP_LIMIT_Mz, BOTTOM_LIMIT_Mz] = adjustAxes('Mz',forceData,StrategyType,TIME_LIMIT_PERC,SIGNAL_THRESHOLD,MARGIN,AVERAGE);
169
170%% Insert State Lines
171            handles         = [pFx pFy pFz pMx pMy pMz];
172
173            % 8 axes for PA10
174            if(~strcmp(StrategyType,'HSA') && ~strcmp(StrategyType,'ErrorCharac')) % Include limits for the rotational spring
175                TOP_LIMIT       = [TOP_LIMIT_S1 TOP_LIMIT_S2 TOP_LIMIT_Fx TOP_LIMIT_Fy TOP_LIMIT_Fz TOP_LIMIT_Mx TOP_LIMIT_My TOP_LIMIT_Mz];
176                BOTTOM_LIMIT    = [BOTTOM_LIMIT_S1 BOTTOM_LIMIT_S2 BOTTOM_LIMIT_Fx BOTTOM_LIMIT_Fy BOTTOM_LIMIT_Fz BOTTOM_LIMIT_Mx BOTTOM_LIMIT_My BOTTOM_LIMIT_Mz];
177            % 6 axes for HIRO
178            else
179                TOP_LIMIT       = [TOP_LIMIT_Fx TOP_LIMIT_Fy TOP_LIMIT_Fz TOP_LIMIT_Mx TOP_LIMIT_My TOP_LIMIT_Mz];
180                BOTTOM_LIMIT    = [BOTTOM_LIMIT_Fx BOTTOM_LIMIT_Fy BOTTOM_LIMIT_Fz BOTTOM_LIMIT_Mx BOTTOM_LIMIT_My BOTTOM_LIMIT_Mz];
181            end
182            % Call insertStates
183            EndTime = forceData(length(forceData),1);   % Pass the last time element of task as endtime.
184            insertStates3(StrategyType,stateData,EndTime,handles,TOP_LIMIT,BOTTOM_LIMIT);    
185
186%% Save plot to file
187            savePlot(fPath,StratTypeFolder,FolderName,handles(1),mfilename);
188    else
189        handles     =-1;
190        TOP_LIMIT   =-1;
191        BOTTOM_LIMIT=-1;
192        
193    end % END DB_PRINT.
194end     % End the function
