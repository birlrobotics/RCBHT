1%%*************************** Documentation *******************************
 2% Function used for Snap Assembly strategies. 
 3% Assigns appropriate strategy path based on the kind of approach used.
 4%
 5% If new categories are created for HIRO, please do a global
 6% search for HSA and make sure that any string comparisons include the new
 7% categories. Need to improve the way we do this. There are several
 8% instances in the code, where parameters change according to the type of
 9% experiment that is being run.
10%
11% I.e. loadData.m, SnapData3.m, InsertStates3.m, WritePrimitivesToFile.m,
12% cleanUp.m, WriteCompositesToFile.m, plotMotionCompositions.m,
13% GradientClassification.m, CustomizePlotLength.m
14%**************************************************************************
15function StratTypeFolder = AssignDir(StrategyType)
16
17% Assign a directory path based on the StrategyType used. 
18    if strcmp(StrategyType,'S')
19        StratTypeFolder = 'PositionControl/StraightLineApproach-NewIKinParams/';            % Straight Line with new IKin params
20    elseif strcmp(StrategyType,'SN')
21        StratTypeFolder = 'PositionControl/StraightLineApproach-NewIkinParams-Noise/';      % Straight Line with new IKin params with noise
22    elseif strcmp(StrategyType,'P')
23        StratTypeFolder = 'PositionControl/PivotApproach-NewIkinParams/';                   % Pivot approach with new IKin Params
24    elseif strcmp(StrategyType,'PN')
25        StratTypeFolder = 'PositionControl/PivotApproach-NewIKin-Noise/';                   % Pivot approach with new IKin Params with noise
26    elseif strcmp(StrategyType,'FS')
27        StratTypeFolder = 'ForceControl/StraightLineApproach/';                             % Used with PA10 Simulation
28    elseif strcmp(StrategyType,'FP')
29        StratTypeFolder = 'ForceControl/PivotApproach/';                                    % Used with PA10 PivotApproach Simulation
30    elseif strcmp(StrategyType,'HSA')
31        StratTypeFolder = 'ForceControl/SideApproach/';                                     % Used with HIRO SideApproach Simulation and Physical
32    elseif strcmp(StrategyType, 'ErrorCharac')
33        StratTypeFolder = 'ForceControl/ErrorCharac/';                                      % Used with HIRO SideApproach to compute error characteristics
34    else
35        StratTypeFolder = '';
36%        FolderName='';
37    end
38end
